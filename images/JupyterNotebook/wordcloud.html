# wordcloud graph
from wordcloud import WordCloud
words = []
for rule in results:
    for item in rule.ordered_statistics[0].items_base:
        if isinstance(item, str):
            words += item.split()
    for item in rule.ordered_statistics[0].items_add:
        if isinstance(item, str):
            words += item.split()

freq = {}
for word in words:
    if word not in freq:
        freq[word] = 1
    else:
        freq[word] += 1


wordcloud = WordCloud(width=600, height=450, 
                      background_color='white').generate_from_frequencies(freq)
plt.figure(figsize=(12, 9))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis('off')
plt.show()
